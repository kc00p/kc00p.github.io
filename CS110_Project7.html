import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Assuming 'Blood Sugar' is a numerical column and 'Heart Attack' is a binary categorical column (e.g., 0 or 1)

# Let's create bins for blood sugar levels
blood_sugar_bins = [0, 100, 140, 200, np.inf]  # Example bins (adjust based on data distribution and medical context)
blood_sugar_labels = ['Normal', 'Pre-diabetes', 'Diabetes', 'High-diabetes']

india_df['Blood_Sugar_Category'] = pd.cut(india_df['Blood_Sugar'], bins=blood_sugar_bins, labels=blood_sugar_labels, right=False)

# Group by the new blood sugar categories and the 'Heart Attack' column
heart_attack_by_blood_sugar = india_df.groupby('Blood_Sugar_Category')['Heart_Attack'].value_counts().unstack().fillna(0)

# If 'Heart Attack' column has values like 0 and 1, rename columns for clarity
# 1 means Heart Attack occurred
heart_attack_by_blood_sugar = heart_attack_by_blood_sugar.rename(columns={0: 'No Heart Attack', 1: 'Heart Attack'})

# Create the bar plot
ax = heart_attack_by_blood_sugar.plot(kind='bar', figsize=(10, 6))

plt.title('Relationship between Blood Sugar Category and Heart Attack Cases')
plt.xlabel('Blood Sugar Category')
plt.ylabel('Number of Patients')
plt.xticks(rotation=45, ha='right')
plt.legend(title='Heart Attack')
plt.tight_layout() # Adjust layout to prevent labels overlapping
plt.show()
